def reverse_ascending(items: list[int]) -> Iterable[int]:

    buffer = []
    for i, el in enumerate(items):
        buffer.append(el)
        if i == len(items)-1 or el >= items[i + 1]:
            yield from reversed(buffer)
            buffer = []
